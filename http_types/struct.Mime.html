Struct http_types::Mime

source · 
pub struct Mime { /* private fields */ }
Expand description

An IANA media type.

use http_types::Mime;
use std::str::FromStr;

let mime = Mime::from_str("text/html;charset=utf-8").unwrap();
assert_eq!(mime.essence(), "text/html");
assert_eq!(mime.param("charset").unwrap(), "utf-8");
